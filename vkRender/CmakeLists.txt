cmake_minimum_required(VERSION 3.20)

project(vkRender LANGUAGES CXX)

file(GLOB_RECURSE SRC_FILES *.cpp)

file(GLOB_RECURSE HEADER_FILES *.h)

add_library(vkRender STATIC
    ${SRC_FILES}
    ${HEADER_FILES}
)

target_compile_features(vkRender PUBLIC cxx_std_20)
target_compile_definitions(vkRender PUBLIC VK_NO_PROTOTYPES)

# Header-only third-party libraries: glm / spdlog / stb_image
target_include_directories(vkRender
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${CMAKE_SOURCE_DIR}/external
        ${CMAKE_SOURCE_DIR}/external/glm
        ${CMAKE_SOURCE_DIR}/external/spdlog
        ${CMAKE_SOURCE_DIR}/external/stb_image
)

target_link_libraries(vkRender
    PUBLIC
        glfw
        volk
)
