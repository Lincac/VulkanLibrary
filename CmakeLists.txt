cmake_minimum_required(VERSION 3.20)
project(VulkanLibrary LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# --- Vulkan SDK ---
find_package(Vulkan REQUIRED)

# --- GLFW ---
add_subdirectory(external/glfw)

# --- volk (optional but recommended) ---
add_subdirectory(external/volk)

# --- Vulkan Renderer ---
add_subdirectory(vkRender)

# --- Source files ---
file(GLOB_RECURSE SRC_FILES src/*.cpp src/*.h)

add_executable(VulkanLibrary ${SRC_FILES})

target_include_directories(VulkanLibrary
    PRIVATE
        src
)

target_link_libraries(VulkanLibrary PRIVATE
    vkRender
)
