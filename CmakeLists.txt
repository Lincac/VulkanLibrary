cmake_minimum_required(VERSION 3.20)
project(VulkanLibrary LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# --- Dependencies ---
include(cmake/Dependencies.cmake)

# --- GLFW ---
add_subdirectory(external/glfw)

# --- volk (optional but recommended) ---
add_subdirectory(external/volk)

# --- Vulkan Renderer ---
add_subdirectory(vkRender)

# --- Source files ---
file(GLOB_RECURSE SRC_FILES src/*.cpp src/*.h)

# --- Shader compilation (shaders -> shaders/bin/*.spv) ---
find_program(GLSLC_EXECUTABLE glslc HINTS ENV VULKAN_SDK PATH_SUFFIXES Bin)
if (NOT GLSLC_EXECUTABLE)
    message(FATAL_ERROR "glslc not found. Please install Vulkan SDK and ensure glslc is in PATH.")
endif()

set(SHADER_SOURCE_DIR ${CMAKE_SOURCE_DIR}/shaders)
set(SHADER_OUTPUT_DIR ${CMAKE_SOURCE_DIR}/shaders/bin)

file(GLOB SHADER_SOURCES CONFIGURE_DEPENDS
    ${SHADER_SOURCE_DIR}/*.vert
    ${SHADER_SOURCE_DIR}/*.frag
    ${SHADER_SOURCE_DIR}/*.comp
    ${SHADER_SOURCE_DIR}/*.geom
    ${SHADER_SOURCE_DIR}/*.tesc
    ${SHADER_SOURCE_DIR}/*.tese
)

set(SHADER_SPV_OUTPUTS)
foreach(SHADER_FILE ${SHADER_SOURCES})
    get_filename_component(SHADER_NAME ${SHADER_FILE} NAME)
    set(SHADER_SPV ${SHADER_OUTPUT_DIR}/${SHADER_NAME}.spv)
    list(APPEND SHADER_SPV_OUTPUTS ${SHADER_SPV})

    add_custom_command(
        OUTPUT ${SHADER_SPV}
        COMMAND ${CMAKE_COMMAND} -E make_directory ${SHADER_OUTPUT_DIR}
        COMMAND ${GLSLC_EXECUTABLE} ${SHADER_FILE} -o ${SHADER_SPV}
        DEPENDS ${SHADER_FILE}
        COMMENT "Compiling shader ${SHADER_NAME} -> ${SHADER_NAME}.spv"
        VERBATIM
    )
endforeach()

add_custom_target(compile_shaders ALL DEPENDS ${SHADER_SPV_OUTPUTS})

add_executable(VulkanLibrary ${SRC_FILES})

target_include_directories(VulkanLibrary
    PRIVATE
        src
)

target_link_libraries(VulkanLibrary PRIVATE
    vkRender
)

add_dependencies(VulkanLibrary compile_shaders)
